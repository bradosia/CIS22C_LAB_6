156 template <class T, class N>
157 void BST<T, N>::addHelper (BSTNode<T, N> *root, N* val, T (*access)(N*))
158 {
159  if ((*access)(root->getValue ()) > (*access)(val))
160  {
161  // val less than root to the left
162  if (!root->getLeft ()) root->setLeft (new BSTNode<T, N> (val));
163  else addHelper (root->getLeft (), val, access);
164  }
165  else
166  {
167  // val greater or equal to root to the right
168  if (!root->getRight ()) root->setRight (new BSTNode<T, N> (val));
169  else addHelper (root->getRight (), val, access);
170  }
171 }

244 template <class T, class N>
245 bool BST<T, N>::removeHelper (BSTNode<T, N>* parent, BSTNode<T, N>* current, T value, T (*access)(N*))
246 {
247  if (!current) return false;
248  if ((*access)(current->getValue ()) == value)
249  {
250  if (current->getLeft () == NULL || current->getRight () == NULL)
251  {
252  BSTNode<T, N>* temp = current->getLeft ();
253  if (current->getRight ()) temp = current->getRight ();
254  if (parent)
255  {
256  if (parent->getLeft () == current)
257  {
258  parent->setLeft (temp);
259  }
260  else
261  {
262  parent->setRight (temp);
263  }
264  }
265  else
266  {
267  this->root = temp;
268  }
269  }
270  else
271  {
272  BSTNode<T, N>* validSubs = current->getRight ();
273  while (validSubs->getLeft ())
274  {
275  validSubs = validSubs->getLeft ();
276  }
277  N* temp = current->getValue ();
278  current->setValue (validSubs->getValue ());
279  validSubs->setValue (temp);
280  return removeHelper (current, current->getRight (), (*access)(temp), access);
281  }
282  delete current;
283  return true;
284  }
285  return removeHelper (current, current->getLeft (), value, access) ||
286  removeHelper (current, current->getRight (), value, access);
287 }

290 void BST<T, N>::findHelper (BSTNode<T, N>* current, T value, List<N*>* listPtr, T (*access)(N*), int &operations)
291 {
292  /*
293  searches the normal add path to find matching values
294  list will be empty if no matches found assuming list was
295  empty in the beginning
296  */
297  if (!current) return;
298  if ((*access)(current->getValue ()) == value)
299  listPtr->push_back (current->getValue ());
300  if ((*access)(current->getValue ()) > value)
301  {
302  // val less than root to the left
303  operations++;
304  findHelper (current->getLeft (), value, listPtr, access, operations);
305  }
306  else
307  {
308  // val greater or equal to root to the right
309  operations++;
310  findHelper (current->getRight (), value, listPtr, access, operations);
311  }
312 }
